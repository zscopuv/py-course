### Updated & Improved Expense Tracker

```python
import json
from datetime import datetime
from typing import List

class Expense:
    def __init__(self, amount: float, category: str, description: str):
        self.amount = amount
        self.category = category
        self.description = description
        self.date = datetime.now().strftime("%Y-%m-%d %H:%M")
    
    def to_dict(self):
        return {
            "amount": self.amount,
            "category": self.category,
            "description": self.description,
            "date": self.date
        }
    
    @classmethod
    def from_dict(cls, data):
        expense = cls(data["amount"], data["category"], data["description"])
        expense.date = data["date"]
        return expense


class ExpenseTracker:
    def __init__(self):
        self.expenses: List[Expense] = []
        self.filename = "expenses.json"
        self.load_expenses()
    
    def add_expense(self, amount, category, description):
        expense = Expense(amount, category, description)
        self.expenses.append(expense)
        print(f"✅ {amount} € added under '{category}'!")
    
    def show_all(self):
        if not self.expenses:
            print("No expenses yet.")
            return
        
        print("\n" + "="*70)
        print(f"{'Date':<20} {'Category':<15} {'Amount':<10} Description")
        print("="*70)
        
        for exp in self.expenses:
            print(f"{exp.date:<20} {exp.category:<15} {exp.amount:<10} € {exp.description}")
        
        total = sum(exp.amount for exp in self.expenses)
        print("="*70)
        print(f"Total Spent: {total:.2f} €")
    
    def show_by_category(self):
        if not self.expenses:
            print("No expenses yet.")
            return
        
        from collections import defaultdict
        category_total = defaultdict(float)
        
        for exp in self.expenses:
            category_total[exp.category] += exp.amount
        
        print("\n" + "="*40)
        print("EXPENSES BY CATEGORY")
        print("="*40)
        for cat, total in sorted(category_total.items()):
            print(f"{cat:<15} : {total:.2f} €")
        print("="*40)
    
    def monthly_summary(self):
        if not self.expenses:
            print("No expenses yet.")
            return
        
        from collections import defaultdict
        monthly = defaultdict(float)
        
        for exp in self.expenses:
            month = exp.date[:7]  # e.g., "2026-05"
            monthly[month] += exp.amount
        
        print("\n" + "="*40)
        print("MONTHLY SUMMARY")
        print("="*40)
        for month, total in sorted(monthly.items()):
            print(f"{month} : {total:.2f} €")
    
    def save_expenses(self):
        with open(self.filename, "w") as file:
            data = [exp.to_dict() for exp in self.expenses]
            json.dump(data, file, indent=4)
    
    def load_expenses(self):
        try:
            with open(self.filename, "r") as file:
                data = json.load(file)
                self.expenses = [Expense.from_dict(item) for item in data]
        except FileNotFoundError:
            self.expenses = []
```

### Enhanced Main Menu

```python
def main():
    tracker = ExpenseTracker()
    
    while True:
        print("\n" + "="*45)
        print("     💰 PERSONAL EXPENSE TRACKER")
        print("="*45)
        print("1. Add New Expense")
        print("2. Show All Expenses")
        print("3. Show by Category")
        print("4. Monthly Summary")
        print("5. Exit")
        print("="*45)
        
        choice = input("\nEnter your choice (1-5): ").strip()
        
        if choice == "1":
            try:
                amount = float(input("Enter amount (€): "))
                category = input("Category (Food, Transport, Rent, etc.): ").strip()
                desc = input("Description: ").strip()
                tracker.add_expense(amount, category, desc)
                tracker.save_expenses()
            except ValueError:
                print("❌ Please enter a valid amount!")
                
        elif choice == "2":
            tracker.show_all()
            
        elif choice == "3":
            tracker.show_by_category()
            
        elif choice == "4":
            tracker.monthly_summary()
            
        elif choice == "5":
            tracker.save_expenses()
            print("👋 Thank you for using Expense Tracker! Goodbye.")
            break
            
        else:
            print("Invalid option. Please choose 1-5.")

if __name__ == "__main__":
    main()
```
